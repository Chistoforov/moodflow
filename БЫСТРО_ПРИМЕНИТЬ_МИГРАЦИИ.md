## 1️⃣ Первый SQL (создание таблицы)

```sql
CREATE TABLE IF NOT EXISTS public.monthly_analytics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  year INTEGER NOT NULL,
  month INTEGER NOT NULL,
  week_number INTEGER NOT NULL,
  days_analyzed INTEGER NOT NULL,
  analysis_text TEXT NOT NULL,
  general_impression TEXT,
  positive_trends TEXT,
  decline_reasons TEXT,
  recommendations TEXT,
  reflection_directions TEXT,
  perplexity_request_id TEXT,
  perplexity_response JSONB,
  is_final BOOLEAN DEFAULT false,
  status TEXT DEFAULT 'completed' CHECK (status IN ('pending', 'processing', 'completed', 'failed')),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  UNIQUE(user_id, year, month, week_number)
);

CREATE INDEX IF NOT EXISTS idx_monthly_analytics_user_date ON public.monthly_analytics(user_id, year, month);
CREATE INDEX IF NOT EXISTS idx_monthly_analytics_final ON public.monthly_analytics(user_id, year, month, is_final) WHERE is_final = true;

ALTER TABLE public.monthly_analytics ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can read their own monthly analytics" ON public.monthly_analytics FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "Service role can manage monthly analytics" ON public.monthly_analytics FOR ALL USING (auth.jwt()->>'role' = 'service_role');
CREATE POLICY "Users can insert their own monthly analytics" ON public.monthly_analytics FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Admins can read all monthly analytics" ON public.monthly_analytics FOR SELECT USING (EXISTS (SELECT 1 FROM public.psychologists WHERE user_id = auth.uid() AND role = 'admin' AND active = true));

CREATE OR REPLACE FUNCTION update_monthly_analytics_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_monthly_analytics_updated_at
  BEFORE UPDATE ON public.monthly_analytics
  FOR EACH ROW
  EXECUTE FUNCTION update_monthly_analytics_updated_at();
```

## 2️⃣ Второй SQL (права для админов)

```sql
CREATE POLICY "Admins can insert monthly analytics for any user"
  ON public.monthly_analytics FOR INSERT
  WITH CHECK (EXISTS (SELECT 1 FROM public.psychologists WHERE user_id = auth.uid() AND role = 'admin' AND active = true));

CREATE POLICY "Admins can update monthly analytics"
  ON public.monthly_analytics FOR UPDATE
  USING (EXISTS (SELECT 1 FROM public.psychologists WHERE user_id = auth.uid() AND role = 'admin' AND active = true));

CREATE POLICY "Admins can delete monthly analytics"
  ON public.monthly_analytics FOR DELETE
  USING (EXISTS (SELECT 1 FROM public.psychologists WHERE user_id = auth.uid() AND role = 'admin' AND active = true));
```
